#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>


int f(int n) {

	if(n-1==0){
		return n;
	}
	else {
		return f(n - 1) + f(n - 2);
	}
	
}


int main(void) {
	
	int n;
	scanf("%d", &n);

	switch (n) {
	case 0 :
	case 1 :
		printf("%d", n);
		break;
	default :
		printf("%d", f(n));
		break;
	}

	return 0;

}
